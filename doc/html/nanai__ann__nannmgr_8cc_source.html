313  nanmath::nanmath_matrix nanai_ann_nannmgr::merge_delta_matrix(nanmath::nanmath_matrix &dmat1,
314  nanmath::nanmath_matrix &dmat2) {
315 
316  if (dmat1.row_size() == dmat2.row_size()) {
317  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);
318  }
319 
320  if (dmat1.col_size() == dmat2.col_size()) {
321  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);
322  }
323 
324  nanmath::nanmath_matrix c(dmat1.row_size(), dmat1.col_size());
325  for (size_t i = 0; i < dmat1.row_size(); i++) {
326  for (size_t j = 0; j < dmat1.col_size(); j++) {
327  /* 偏差越小在融合后的矩阵中所占值越大 */
328  double delta_a = 1 - (dmat1[i][j] / dmat1[i][j] + dmat2[i][j]);
329  double delta_b = 1 - (dmat2[i][j] / dmat1[i][j] + dmat2[i][j]);
330 
331  double c_n = 1/2 * (delta_a * dmat1[i][j] + delta_b * dmat2[i][j]);
332  c.set(i, j, c_n);
333  }
334  }
335 
336  return c;
337  }

371  nanai_ann_nanncalc::ann_t nanai_ann_nannmgr::merge_ann(nanai_ann_nanncalc::ann_t &a,
372  nanai_ann_nanncalc::ann_t &b) {
373  nanai_ann_nanncalc::ann_t c;
374  if (a.weight_matrixes.size() != b.weight_matrixes.size()) {
375  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);

387  for (size_t i = 0; i < a.weight_matrixes.size(); i++) {
388  wm.push_back(merge_matrix(a.weight_matrixes[i], b.weight_matrixes[i],
389  a.delta_weight_matrixes[i], b.delta_weight_matrixes[i]));
390  dwm.push_back(merge_delta_matrix(a.delta_weight_matrixes[i], b.delta_weight_matrixes[i]));
391  }
392  c.weight_matrixes = wm;
393  c.delta_weight_matrixes = dwm;

568  nanai_ann_nanncalc *nanai_ann_nannmgr::generate(nanai_ann_nanndesc &desc,
569  nanai_ann_nanncalc::ann_t *ann,
570  const char *task) {
571  lock();
572  nanai_ann_nanncalc *calc = nullptr;
573  if (_curr_calc >= _max_calc) {
574 
575  /* 从所有线程中找到相同任务名的所有结点 */
576  std::vector<nanai_ann_nanncalc*> found_node;
577 
578  /* 从所有线程中找到相同任务名的所有结点 */
579  for (auto i : _calcs) {
580  if (i->get_task_name() == task) {
581  found_node.push_back(i);
582  }
583  }
584 
585  if (found_node.empty() == false) {
586  /* 找到了，则找一个命令数量最多的
587  * 这里的目的是延迟计算，让同等任务名的计算优先计算完毕
588  */
589  std::sort(found_node.begin(), found_node.end(),
590  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
591  return a->get_cmdlist_count() > b->get_cmdlist_count();
592  });
593  /* 这里可能也会造成算法更替，所以这里替换算法保障计算正确 */
594  found_node[0]->ann_configure(desc);
595  } else {/* 如果没有找到则进行策略2 */
596  found_node.clear();
597  /* 从所有线程中找到相同算法的所有结点 */
598  for (auto i : _calcs) {
599  if (i->get_alg_name() == desc.name) {
600  found_node.push_back(i);
601  }
602  }
603 
604  /* 如果没有找到则直接选定一个命令数量最小的，进行重新配置 */
605  if (found_node.empty()) {
606  std::sort(_calcs.begin(), _calcs.end(),
607  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
608  return a->get_cmdlist_count() < b->get_cmdlist_count();
609  });
610  found_node.push_back(_calcs[0]);
611  found_node[0]->ann_configure(desc);
612  } else {
613  /* 从已经找到的结点中寻找一个命令数量最少的 */
614  std::sort(found_node.begin(), found_node.end(),
615  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
616  return a->get_cmdlist_count() < b->get_cmdlist_count();
617  });
618  }
619  }
620 
621  calc = found_node[0];
622  } else {
623  calc = make(desc);
624  _curr_calc++;
625  }
