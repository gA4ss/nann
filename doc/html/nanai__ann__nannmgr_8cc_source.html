377  nanmath::nanmath_matrix nanai_ann_nannmgr::merge_delta_matrix(nanmath::nanmath_matrix &dmat1,
378  nanmath::nanmath_matrix &dmat2) {
379 
380  if (dmat1.row_size() == dmat2.row_size()) {
381  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);
382  }
383 
384  if (dmat1.col_size() == dmat2.col_size()) {
385  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);
386  }
387 
388  nanmath::nanmath_matrix c(dmat1.row_size(), dmat1.col_size());
389  for (size_t i = 0; i < dmat1.row_size(); i++) {
390  for (size_t j = 0; j < dmat1.col_size(); j++) {
391  /* 偏差越小在融合后的矩阵中所占值越大 */
392  double delta_a = 1 - (dmat1[i][j] / dmat1[i][j] + dmat2[i][j]);
393  double delta_b = 1 - (dmat2[i][j] / dmat1[i][j] + dmat2[i][j]);
394 
395  double c_n = 1/2 * (delta_a * dmat1[i][j] + delta_b * dmat2[i][j]);
396  c.set(i, j, c_n);
397  }
398  }
399 
400  return c;
401  }

436  nanai_ann_nanncalc::ann_t &b) {
437  nanai_ann_nanncalc::ann_t c;
438  if (a.weight_matrixes.size() != b.weight_matrixes.size()) {
439  error(NANAI_ERROR_LOGIC_INVALID_ARGUMENT);

445 
446  c = a;

453  a.delta_weight_matrixes[i], b.delta_weight_matrixes[i]));
454  dwm.push_back(merge_delta_matrix(a.delta_weight_matrixes[i], b.delta_weight_matrixes[i]));
455  }

572  void nanai_ann_nannmgr::get_algs(std::string &path) {
573  get_def_algs();
574 
575  /* 遍历_alg_dir下的所有so文件并加载
576  * 并且通过通用接口 "ann_alg_setup"
577  * 如果没有此接口则不是算法控件
578  */
579 
580  if (path.back() != '/') {
581  path += '/';
582  }
583 
584  DIR *dir;
585  struct dirent *ent;;
586  std::string filename, subdir;
587 
588  dir = opendir(path.c_str());
589 
590  while ((ent = readdir(dir))) {
591  if (ent->d_type & DT_DIR) {
592  if (strcmp(ent->d_name,".") == 0 || strcmp(ent->d_name, "..") == 0)
593  continue;
594 
595  if (1) {
596  subdir = path + ent->d_name;
597  get_algs(subdir);
598  }
599  } else {
600  /* 加载so文件 */
601  filename = path + ent->d_name;
602  void *h = dlopen(filename.c_str(), RTLD_NOW);
603  if (h) {
604  fptr_ann_alg_setup pf = (fptr_ann_alg_setup)dlsym(h, "ann_alg_setup");
605  if (pf) {
606  /* 安装 */
607  if (add_alg(*pf(_etc_dir.c_str()))) {
608  auto v = std::make_pair(h, pf);
609  _algs.push_back(v);
610  }
611  }/* end if */
612  }
613  }/* end else */
614  }/* end while */
615

661  nanai_ann_nanncalc *nanai_ann_nannmgr::generate(nanai_ann_nanndesc &desc,
662  nanai_ann_nanncalc::ann_t *ann,
663  const char *task) {
664  lock();
665  nanai_ann_nanncalc *calc = nullptr;
666  if (_curr_calc >= _max_calc) {
667 
668  /* 从所有线程中找到相同任务名的所有结点 */
669  std::vector<nanai_ann_nanncalc*> found_node;
670 
671  /* 从所有线程中找到相同任务名的所有结点 */
672  for (auto i : _calcs) {
673  if (i->get_task_name() == task) {
674  found_node.push_back(i);
675  }
676  }
677 
678  if (found_node.empty() == false) {
679  /* 找到了，则找一个命令数量最多的
680  * 这里的目的是延迟计算，让同等任务名的计算优先计算完毕
681  */
682  std::sort(found_node.begin(), found_node.end(),
683  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
684  return a->get_cmdlist_count() > b->get_cmdlist_count();
685  });
686  /* 这里可能也会造成算法更替，所以这里替换算法保障计算正确 */
687  found_node[0]->ann_configure(desc);
688  } else {/* 如果没有找到则进行策略2 */
689  found_node.clear();
690  /* 从所有线程中找到相同算法的所有结点 */
691  for (auto i : _calcs) {
692  if (i->get_alg_name() == desc.name) {
693  found_node.push_back(i);
694  }
695  }
696 
697  /* 如果没有找到则直接选定一个命令数量最小的，进行重新配置 */
698  if (found_node.empty()) {
699  std::sort(_calcs.begin(), _calcs.end(),
700  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
701  return a->get_cmdlist_count() < b->get_cmdlist_count();
702  });
703  found_node.push_back(_calcs[0]);
704  found_node[0]->ann_configure(desc);
705  } else {
706  /* 从已经找到的结点中寻找一个命令数量最少的 */
707  std::sort(found_node.begin(), found_node.end(),
708  [](nanai_ann_nanncalc *a, nanai_ann_nanncalc *b) {
709  return a->get_cmdlist_count() < b->get_cmdlist_count();
710  });
711  }
712  }
713 
714  calc = found_node[0];
715  } else {
716  calc = make(desc, task);
717  _curr_calc++;
718  }
719 
720  /* 替换神经网络 */
721  if (ann) {
722  calc->do_ann_exchange(*ann);
723  }
724 
725  unlock();
726 
727  return calc;
728  }
