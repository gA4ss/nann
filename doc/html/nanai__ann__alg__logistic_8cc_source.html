54 void ann_hidden_adjust_weight(int h,
55  nanmath::nanmath_vector *layer,
56  nanmath::nanmath_vector *delta,
57  nanmath::nanmath_matrix *wm,
58  nanmath::nanmath_matrix *prev_dwm) {
59  /* 这里是遍历列向量 */
60  for (size_t i = 0; i < delta->size(); i++) { /* 矩阵的列 */
61  for (size_t j = 0; j < layer->size(); j++) { /* 矩阵的行 */
62  /* 让上一层的每个输入向量都乘以当前的偏差值
63  * 然后在修订这个偏差值的权向量
64  */
65  double new_dw = (g_eta * delta->at(i) * layer->at(j)) + (g_momentum * prev_dwm->at(j, i));
66  double t = wm->at(j, i) + new_dw;
67  wm->set(j, i, t);
68  prev_dwm->set(j, i, new_dw);
69  }
70  }
71 }

150  std::ifstream file;
151  file.open(filename, std::ios::in|std::ios::binary);
152  if (file.is_open() == false) {
153  nanai::error(NANAI_ERROR_RUNTIME_OPEN_FILE);
154  }
155 
156  /* 映射文件 */
157  file.seekg(0, std::ios::end);
158  long long filesize = file.tellg();
159 
160  char *buf = new char [filesize + 1];
161  if (buf == nullptr) {
162  nanai::error(NANAI_ERROR_RUNTIME_ALLOC_MEMORY);
163  }
164 
165  file.seekg(0, std::ios::beg);
166  file.read((char*)buf, filesize);
167  cJSON *json = parse_conf(buf);
168  file.close();
169  delete [] buf;
170 
171  return json;
