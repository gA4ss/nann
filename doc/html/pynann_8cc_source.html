108 static int parse_create_json_read_matrix(cJSON *json,
109  nanmath::nanmath_matrix &matrix) {
110  int nrow = 0, ncol = 0, nprev = 0;
111  cJSON *row = json, *col = nullptr;
112  std::vector<double> row_v;
113  matrix.clear();
114 
115  while (row) {
116  col = row->child;
117  if (col == nullptr) {
118  return PYNANN_ERROR_PARSE_JSON;
119  }
120 
121  ncol = 0;
122  row_v.clear();
123  while (col) {
124  if (col->valuedouble == 0.0) row_v.push_back((double)col->valueint);
125  else row_v.push_back(col->valuedouble);
126  ncol++;
127  col = col->next;
128  }
129 
130  if (nprev) {
131  /* 每列必须一样的数量 */
132  if (nprev != ncol) {
133  matrix.clear();
134  return PYNANN_ERROR_PARSE_JSON;
135  }
136  }
137  nprev = ncol;
138 
139  /* 压入一行 */
140  matrix.push_row(row_v);
141 
142  nrow++;
143  row = row->next;
144  }
145 
146  return 0;
147 }

199 static int parse_create_json(cJSON *json,
200  std::string &alg,
201  nanai_ann_nanncalc::ann_t &ann,
202  nanmath::nanmath_vector &target) {
203  int ret = PYNANN_ERROR_SUCCESS;
204  if (json == nullptr) {
205  return PYNANN_ERROR_INTERNAL;
206  }
207  cJSON *json_child = json->child;
208  if (json_child == nullptr) {
209  return PYNANN_ERROR_PARSE_JSON;
210  }
211 
212  /* 遍历根结点 */
213  bool handle_alg = false, handle_ann = false;
214  while (json_child) {
215  if (strcmp(json_child->string, "alg") == 0) {
216  alg = json_child->valuestring;
217  handle_alg = true;
218  } else if (strcmp(json_child->string, "ann") == 0) {
219  cJSON *json_next = json_child->child;
220  size_t wm_ninput = 0, dwm_ninput = 0,
221  wm_nhidden = 0, dwm_nhidden = 0,
222  wm_noutput = 0, dwm_noutput = 0;
223 
224  std::vector<int> wm_nneure, dwm_nneure;
225  while (json_next) {
226  if (strcmp(json_next->string, "weight matrixes") == 0) {
227  if ((ret = parse_create_json_matrixes(json_next->child,
228  wm_ninput,
229  wm_nhidden,
230  wm_noutput,
231  ann.weight_matrixes)) != PYNANN_ERROR_SUCCESS) {
232  goto _error;
233  }
234  } else if (strcmp(json_next->string, "delta weight matrixes") == 0) {
235  if ((ret = parse_create_json_matrixes(json_next->child,
236  dwm_ninput,
237  dwm_nhidden,
238  dwm_noutput,
239  ann.delta_weight_matrixes)) != PYNANN_ERROR_SUCCESS) {
240  goto _error;
241  }
242  }
243  json_next = json_next->next;
244  }/* end while */
245  handle_ann = true;
246 
247  if (ann.weight_matrixes.empty()) {
248  ret = PYNANN_ERROR_PARSE_JSON;
249  goto _error;
250  }
251 
252  /* 偏差权值矩阵不为空 */
253  if (ann.delta_weight_matrixes.empty() == false) {
254  if ((wm_ninput != dwm_ninput) ||
255  (wm_nhidden != dwm_nhidden) ||
256  (wm_noutput != dwm_noutput) ||
257  (ann.weight_matrixes.size() != ann.delta_weight_matrixes.size())) {
258  ret = PYNANN_ERROR_PARSE_JSON;
259  goto _error;
260  }
261  }/* end if */
262 
263  ann.ninput = wm_ninput;
264  ann.nhidden = wm_nhidden;
265  ann.noutput = wm_noutput;
266 
267  } else if (strcmp(json_child->string, "target") == 0) {
268  cJSON *json_next = json_child->child;
269  if (json_next == nullptr) { ret = PYNANN_ERROR_PARSE_JSON; goto _error; }
270  target.clear();
271  while (json_next) {
272  if (json_next->valuedouble) target.push(json_next->valuedouble);
273  else target.push((double)json_next->valueint);
274  json_next = json_next->next;
275  }
276  }
277 
278  json_child = json_child->next;
279  }/* end while */
280 
281  if ((handle_ann == false) || (handle_alg == false)) {
282  ret = PYNANN_ERROR_PARSE_JSON;
283  goto _error;
284  }
285 
286  if (json) cJSON_Delete(json);
287  return 0;
288 _error:
289  alg.clear();
290  ann.clear();
291  target.clear();
292  return ret;
293 }
