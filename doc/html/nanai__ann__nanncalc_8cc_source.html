560  void nanai_ann_nanncalc::ann_create(nanai_ann_nanndesc &desc) {
561  ann_destroy();
562 
563  _alg = desc.name;
564  _ann.ninput = desc.ninput;
565  _ann.nhidden = (desc.nhidden > MAX_HIDDEN_NUMBER) ? MAX_HIDDEN_NUMBER : desc.nhidden;
566  _ann.noutput = desc.noutput;
567  _output_error.create(_ann.noutput);
568 
569  _fptr_input_filter = desc.fptr_input_filter;
570  _fptr_result = desc.fptr_result;
571  _fptr_output_error = desc.fptr_output_error;
572  _fptr_calculate = desc.fptr_calculate;
573 
574  _fptr_hidden_inits = desc.fptr_hidden_inits;
575  _fptr_hidden_calcs = desc.fptr_hidden_calcs;
576  _fptr_hidden_errors = desc.fptr_hidden_errors;
577  _fptr_hidden_adjust_weights = desc.fptr_hidden_adjust_weights;
578 
579  _callback_monitor_except = desc.callback_monitor_except;
580  _callback_monitor_trained = desc.callback_monitor_trained;
581  _callback_monitor_trained_nooutput = desc.callback_monitor_trained_nooutput;
582  _callback_monitor_calculated = desc.callback_monitor_calculated;
583  _callback_monitor_progress = desc.callback_monitor_progress;
584 
585  /* 创建神经网络 */
586  ann_log("--------------------------------------------------");
587  ann_log("ann start creating");
588  ann_log("algorithm = %s", _alg.c_str());
589  ann_log("ninput = %d", _ann.ninput);
590  ann_log("nhidden = %d", _ann.nhidden);
591  ann_log("noutput = %d", _ann.noutput);
592  ann_log("total of weight matrix = %d", _ann.nhidden+1);
593 
594  _hiddens.resize(_ann.nhidden);
595  _ann.weight_matrixes.resize(_ann.nhidden + 1);
596  _ann.delta_weight_matrixes.resize(_ann.nhidden + 1);
597  for (int i = 0; i < _ann.nhidden + 1; i++) {
598  /* 隐藏层的个数要比需要的权重矩阵个数少1 */
599  if (i < _ann.nhidden) {
600  _ann.nneural.push_back(desc.nneure[i]);/* 是否不需要nneural */
601  _hiddens[i].create(_ann.nneural[i]);
602  ann_log("%dth hidden has %d neurals", i+1, _ann.nneural[i]);
603  }
604 
605  /*
606  * 根据隐藏层所处的位置不同，进行不同的权值矩阵的构造
607  */
608  if (i == 0) {
609  /* 处理第一层，第一层是由输入向量 与 第一个隐藏层之间的权值矩阵 */
610  _ann.weight_matrixes[i].create(_ann.ninput, _ann.nneural[i]);
611  } else if (i == _ann.nhidden) {
612  /* 处理最后一层，最后一层是最后一个隐藏层 与 输出向量之间的权值矩阵
613  * 因为权值矩阵的个数要比隐藏层的个数多一个，所以到达这个时候要获取
614  * 最后一个隐藏层必须退回一个索引
615  */
616  _ann.weight_matrixes[i].create(_ann.nneural[i-1], _ann.noutput);
617  } else {
618  /* 其余是隐藏层 与 隐藏层之间的权值矩阵 */
619  _ann.weight_matrixes[i].create(_ann.nneural[i-1], _ann.nneural[i]);
620  }
621 
622  /* 初始化当前层的权值矩阵 */
623  _ann.delta_weight_matrixes[i].zero();
624  if (_fptr_hidden_inits) {
625  _fptr_hidden_inits(i, &_ann.weight_matrixes[i]);
626  } else {
627  _ann.weight_matrixes[i].random();
628  }
629  }
630 
631  ann_log("ann create successful");
632  }

746  void nanai_ann_nanncalc::ann_forward(nanmath::nanmath_vector &input, nanmath::nanmath_vector &output) {
747  nanmath::nanmath_vector l1;
748  nanmath::nanmath_vector l2;
749  nanmath::nanmath_matrix wm;
750 
751  /* 运算的次数(权值矩阵的个数)要比隐藏层数多1 */
752  for (int i = 0; i < _ann.nhidden+1; i++) {
753  if (i == 0) {
754  l1 = input;
755  l2 = _hiddens[i];
756  wm = _ann.weight_matrixes[i];
757  } else if (i == _ann.nhidden) {
758  l1 = _hiddens[i-1];
759  l2 = output;
760  wm = _ann.weight_matrixes[i];
761  } else {
762  l1 = _hiddens[i-1];
763  l2 = _hiddens[i];
764  wm = _ann.weight_matrixes[i];
765  }
766 
767  if (_fptr_hidden_calcs) {
768  ann_layer_forward(i, l1, l2, wm, _fptr_hidden_calcs);
769  }
770 
771  }
772 
773  output = l2;
774 
775  }

782  void nanai_ann_nanncalc::ann_layer_forward(int h,

832  nanmath::nanmath_vector nanai_ann_nanncalc::ann_calc_hidden_delta(size_t h,
833  nanmath::nanmath_vector &delta_k,
834  nanmath::nanmath_matrix &w_kh,
835  nanmath::nanmath_vector &o_h) {
836  nanmath::nanmath_vector delta_h(o_h.size());

846  void nanai_ann_nanncalc::ann_hiddens_error(nanmath::nanmath_vector &input, nanmath::nanmath_vector &output_delta) {
847  /* 遍历所有隐藏权值矩阵 */
848  nanmath::nanmath_vector delta_k = output_delta;
849  nanmath::nanmath_vector i;
850 
851  /* 隐藏层比权值矩阵少一个 */
852  for (size_t h = _ann.weight_matrixes.size() - 1; h > 0; h--) {
853  /* 所需重要的变量
854  * h 处于第几个权值矩阵运算
855  * delta_k 上一层的误差向量
856  * w_kh h->k的权值矩阵
857  * o_h 当前隐藏单元的h的值
858  *
859  * 目的是求出delta_h，当前隐藏层的相对于上一次的偏差
860  */
861 
862  if (h == 1) {
863  /* 上一层就是输入层 */
864  i = input;
865  } else {
866  i = _hiddens[h-1];
867  }
868 
869  if (_fptr_hidden_errors) {
870  _fptr_hidden_errors((int)h, &delta_k, &_ann.weight_matrixes[h], &i, &delta_k);
871  } else {
872  delta_k = ann_calc_hidden_delta(h, delta_k, _ann.weight_matrixes[h], i);
873  }
874 
875  /* 调整当前的权值矩阵 */
876  if (_fptr_hidden_adjust_weights) {
877  _fptr_hidden_adjust_weights((int)h, &i, &delta_k,
878  &_ann.weight_matrixes[h], &_ann.delta_weight_matrixes[h]);
879  } else {
880  ann_adjust_weight(h, i, delta_k);
881  }
882  }
883  }

885  void nanai_ann_nanncalc::ann_adjust_weight(size_t h,
886  nanmath::nanmath_vector &layer,
887  nanmath::nanmath_vector &delta) {
888  static const double s_eta = 0.05;/* 学习速率 */
889  static const double momentum = 0.03;/* 冲量项 */
890  nanmath::nanmath_matrix &wm = _ann.weight_matrixes[h];
891  nanmath::nanmath_matrix &prev_dwm = _ann.delta_weight_matrixes[h];
892 
893  /* 这里是遍历列向量 */
894  for (int i = 0; i < delta.size(); i++) { /* 矩阵的列 */
895  for (int j = 0; j < layer.size(); j++) { /* 矩阵的行 */
896  /* 让上一层的每个输入向量都乘以当前的偏差值
897  * 然后在修订这个偏差值的权向量
898  */
899  double new_dw = (s_eta * delta[i] * layer[j]) + (momentum * prev_dwm[j][i]);
900  wm[j][i] += new_dw;
901  prev_dwm[j][i] = new_dw;
902  }
903  }/* end for */
904  }
